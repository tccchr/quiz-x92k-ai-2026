-- ========================================================================= -- 🔥 BATTLE SYSTEM v3.1 — แก้ Race Condition ด้วย Advisory Lock -- -- ปัญหาเดิม: 2 คน submit พร้อมกัน → COUNT(*) = 0 ทั้งคู่ → ทั้งคู่ rank 1 -- วิธีแก้: pg_advisory_xact_lock บังคับให้ทำทีละคนต่อข้อ -- ========================================================================= -- 0. ล้าง function เก่าทุก signature DO $$ DECLARE r RECORD; BEGIN FOR r IN SELECT oid::regprocedure::text AS sig FROM pg_proc WHERE proname = 'submit_battle_answer' LOOP EXECUTE 'DROP FUNCTION IF EXISTS ' || r.sig || ' CASCADE'; END LOOP; FOR r IN SELECT oid::regprocedure::text AS sig FROM pg_proc WHERE proname = 'finalize_battle' LOOP EXECUTE 'DROP FUNCTION IF EXISTS ' || r.sig || ' CASCADE'; END LOOP; FOR r IN SELECT oid::regprocedure::text AS sig FROM pg_proc WHERE proname = 'award_battle_winner' LOOP EXECUTE 'DROP FUNCTION IF EXISTS ' || r.sig || ' CASCADE'; END LOOP; END $$; -- 1. สร้างตาราง DROP TABLE IF EXISTS battle_answers CASCADE; CREATE TABLE battle_answers ( id BIGSERIAL PRIMARY KEY, room_id UUID NOT NULL REFERENCES battle_rooms(id) ON DELETE CASCADE, emp_id TEXT NOT NULL, q_idx INT NOT NULL, question_id BIGINT NOT NULL, selected_choice INT NOT NULL, is_correct BOOLEAN NOT NULL DEFAULT false, time_taken INT NOT NULL DEFAULT 0, rank_in_question INT DEFAULT 0, base_points INT NOT NULL DEFAULT 0, speed_bonus INT NOT NULL DEFAULT 0, total_points INT NOT NULL DEFAULT 0, server_timestamp TIMESTAMPTZ NOT NULL DEFAULT now(), UNIQUE(room_id, emp_id, q_idx) ); CREATE INDEX IF NOT EXISTS idx_ba_room ON battle_answers(room_id, q_idx); -- ========================================================================= -- 2. submit_battle_answer — ใช้ ADVISORY LOCK ป้องกัน race condition -- Lock ต่อ (room_id, q_idx) → บังคับให้ทำทีละคนต่อข้อ -- ========================================================================= CREATE OR REPLACE FUNCTION submit_battle_answer( p_emp_id TEXT, p_room_id UUID, p_q_idx INT, p_question_id BIGINT, p_selected INT, p_time_taken INT, p_streak INT DEFAULT 0 ) RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$ DECLARE v_correct_answer TEXT; v_is_correct BOOLEAN; v_rank INT := 0; v_base_points INT := 0; v_speed_bonus INT := 0; v_total_points INT := 0; v_existing RECORD; v_time_seconds NUMERIC; v_correct_before_me INT; v_lock_key BIGINT; BEGIN -- Anti-cheat IF p_time_taken < 300 THEN RETURN jsonb_build_object('success', false, 'message', 'cheat', 'is_correct', false, 'points', 0); END IF; -- 🔒 Advisory Lock ต่อ (room_id, q_idx) — ป้องกัน 2 คน submit พร้อมกัน -- hash room_id + q_idx เป็น bigint เพื่อใช้เป็น lock key v_lock_key := abs(hashtext(p_room_id::text || '_' || p_q_idx::text)); PERFORM pg_advisory_xact_lock(v_lock_key); -- Idempotent check (หลัง lock เพื่อความถูกต้อง) SELECT * INTO v_existing FROM battle_answers WHERE room_id = p_room_id AND emp_id = p_emp_id AND q_idx = p_q_idx; IF FOUND THEN RETURN jsonb_build_object('success', true, 'already_answered', true, 'is_correct', v_existing.is_correct, 'points', v_existing.total_points, 'rank', v_existing.rank_in_question); END IF; -- ตรวจคำตอบ SELECT answer INTO v_correct_answer FROM questions WHERE id = p_question_id; IF v_correct_answer IS NULL THEN RETURN jsonb_build_object('success', false, 'message', 'question_not_found'); END IF; v_is_correct := (p_selected::TEXT = v_correct_answer); IF v_is_correct THEN -- 🔒 ตอนนี้ lock อยู่ → COUNT(*) จะถูกต้อง 100% SELECT COUNT(*) INTO v_correct_before_me FROM battle_answers WHERE room_id = p_room_id AND q_idx = p_q_idx AND is_correct = true; v_rank := v_correct_before_me + 1; -- 1v1: เฉพาะ rank 1 ได้แต้ม IF v_rank = 1 THEN v_base_points := 100; v_time_seconds := p_time_taken::NUMERIC / 1000.0; IF v_time_seconds < 1.0 THEN v_speed_bonus := 20; ELSIF v_time_seconds < 2.0 THEN v_speed_bonus := 18; ELSIF v_time_seconds < 3.0 THEN v_speed_bonus := 15; ELSIF v_time_seconds < 4.0 THEN v_speed_bonus := 12; ELSIF v_time_seconds < 5.0 THEN v_speed_bonus := 9; ELSIF v_time_seconds < 6.0 THEN v_speed_bonus := 6; ELSIF v_time_seconds < 7.0 THEN v_speed_bonus := 3; ELSE v_speed_bonus := 1; END IF; END IF; v_total_points := v_base_points + v_speed_bonus; END IF; -- INSERT (UNIQUE ป้องกันซ้ำ) INSERT INTO battle_answers ( room_id, emp_id, q_idx, question_id, selected_choice, is_correct, time_taken, rank_in_question, base_points, speed_bonus, total_points ) VALUES ( p_room_id, p_emp_id, p_q_idx, p_question_id, p_selected, v_is_correct, p_time_taken, v_rank, v_base_points, v_speed_bonus, v_total_points ); -- Lock จะปล่อยอัตโนมัติเมื่อ transaction จบ RETURN jsonb_build_object( 'success', true, 'already_answered', false, 'is_correct', v_is_correct, 'points', v_total_points, 'rank', v_rank, 'base_points', v_base_points, 'speed_bonus', v_speed_bonus ); END; $$; -- ========================================================================= -- 3. finalize_battle -- ========================================================================= CREATE OR REPLACE FUNCTION finalize_battle( p_room_id UUID, p_emp_id TEXT DEFAULT NULL ) RETURNS JSONB LANGUAGE plpgsql SECURITY DEFINER AS $$ DECLARE v_players JSONB := '[]'::jsonb; v_player RECORD; BEGIN FOR v_player IN ( SELECT emp_id, COUNT(*) FILTER (WHERE is_correct) as correct_count, COALESCE(SUM(total_points), 0) as total_score, COALESCE(SUM(time_taken), 0) as total_time FROM battle_answers WHERE room_id = p_room_id GROUP BY emp_id ORDER BY total_score DESC, total_time ASC ) LOOP v_players := v_players || jsonb_build_object( 'emp_id', v_player.emp_id, 'correct', v_player.correct_count, 'score', v_player.total_score, 'total_time', v_player.total_time); END LOOP; RETURN jsonb_build_object('room_id', p_room_id, 'players', v_players); END; $$; -- ========================================================================= -- 4. award_battle_winner -- ========================================================================= CREATE OR REPLACE FUNCTION award_battle_winner( p_emp_id TEXT, p_room_id UUID, p_personal_pts INT DEFAULT 2000, p_site_pts INT DEFAULT 0 ) RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER AS $$ DECLARE v_winner_id TEXT; v_results JSONB; BEGIN SELECT finalize_battle(p_room_id, p_emp_id) INTO v_results; v_winner_id := v_results->'players'->0->>'emp_id'; IF v_winner_id IS NULL OR v_winner_id != p_emp_id THEN RETURN false; END IF; INSERT INTO logs (employee_id, point, question_id) VALUES (p_emp_id, p_personal_pts, 0); RETURN true; END; $$; -- ========================================================================= -- 5. Permissions & RLS -- ========================================================================= GRANT SELECT, INSERT ON battle_answers TO anon; GRANT USAGE, SELECT ON SEQUENCE battle_answers_id_seq TO anon; GRANT EXECUTE ON FUNCTION submit_battle_answer(TEXT, UUID, INT, BIGINT, INT, INT, INT) TO anon; GRANT EXECUTE ON FUNCTION finalize_battle(UUID, TEXT) TO anon; GRANT EXECUTE ON FUNCTION award_battle_winner(TEXT, UUID, INT, INT) TO anon; ALTER TABLE battle_answers ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "ba_insert" ON battle_answers; DROP POLICY IF EXISTS "ba_select" ON battle_answers; CREATE POLICY "ba_insert" ON battle_answers FOR INSERT TO anon WITH CHECK (true); CREATE POLICY "ba_select" ON battle_answers FOR SELECT TO anon USING (true);
